Workflow for making maps

  • Use synthea_data_ctracts_raw dataset as starting point.
  • starting fields: select(GEOID, patid, …)
  • Summarize to relevant information
  • merge data into durham_gis_base
  • Make the map

Map 1: Basic Map

## Among all program participants from 2015-2020, Count number of unique participants by census tract
patid_count_by_ctract  <- dat |> 
  select(GEOID, patid, programid, completion_date1, enrollment_date1) |> 
  distinct() |> 
  group_by(GEOID) |> 
  summarise(count_patid = n())
## Good

## Merge durham gis data
patid_count_by_ctract_b <- durham_gis_base |>
  left_join(patid_count_by_ctract, by = "GEOID")

## Map of participant count
patid_count_by_ctract_b %>%
  ggplot(aes(fill = factor(count_patid))) + 
  geom_sf() +
  theme_void() + 
  scale_fill_viridis_d("Participant\nCount") +
  labs(
    title = "Unique Program Participants in 2015-2020\nby Census Tract"
  )

Map 2: Map of patients with overweight obesity

## Filter to patients the year they joined the program
patid_count_ow_obes_a  <- dat |> 
  select(GEOID, patid, programid, completion_date1, enrollment_date1, year_enroll, year, BMI) |> 
  ## Get demographics of patient for year they enrolled in program
  filter(year_enroll == year) |> 
  distinct() |> 
  filter(BMI >= 25) |> 
  ### Summarise by census tract
  group_by(GEOID) |> 
  summarise(count_patid = n())
  
## Merge durham gis data
patid_count_ow_obes_b <- durham_gis_base |>
  left_join(patid_count_ow_obes_a, by = "GEOID")

## Map of overweight or obesity counts
patid_count_ow_obes_b %>%
  ggplot(aes(fill = factor(count_patid))) + 
  geom_sf() +
  theme_void() + 
  scale_fill_viridis_d("Participant\nCount") +
  labs(
    title = "Unique Program Participants with Overweight or Obesity \n in 2015-2020 by Census Tract"
  )
